    // Handle environment button selection
    const envButtons = document.querySelectorAll('.env-btn');
    const contextFiles = document.getElementById('context-files');
    const functionTypes = document.getElementById('function-types');

    envButtons.forEach(button => {
        button.addEventListener('click', () => {
            // Remove active class from all buttons
            envButtons.forEach(btn => btn.classList.remove('active'));
            // Add active class to clicked button
            button.classList.add('active');

            const env = button.getAttribute('data-env');
            
            // Show/hide context files and function types based on environment
            if (env === 'production' || env === 'development') {
                contextFiles.classList.remove('hidden');
                functionTypes.classList.remove('hidden');
            } else {
                contextFiles.classList.add('hidden');
                functionTypes.classList.add('hidden');
            }
        });
    });

    // Initially hide context files and function types
    contextFiles.classList.add('hidden');
    functionTypes.classList.add('hidden');
</script>
